## Existing Backend Patterns (from Phase 1)

The project uses this pattern per domain:
- `internal/{domain}/service.go` — business logic, accepts `*pgxpool.Pool`, creates `*sqlc.Queries`
- `internal/{domain}/handler.go` — HTTP handlers with chi router, JSON request/response
- `db/queries/{domain}.sql` — sqlc query definitions
- `db/migrations/XXX_name.up.sql` / `.down.sql` — golang-migrate schema changes
- Services call other services directly via constructor injection (e.g., `NewSalesService(queries, inventoryService)`)

- **D-04 (Locked):** Client-generated UUIDs for offline orders — frontend generates UUIDv4, backend accepts it. Idempotency: duplicate UUID silently succeeds.

       package sales

       import (
           "context"
           "errors"
           "fmt"
           "time"

           "github.com/jackc/pgx/v5/pgtype"
           "github.com/jackc/pgx/v5/pgxpool"

           "github.com/zestzero/openpos/db/sqlc"
           "github.com/zestzero/openpos/internal/inventory"
       )

       var (
           ErrOrderNotFound    = errors.New("order not found")
           ErrInvalidOrder     = errors.New("invalid order")
           ErrEmptyOrder       = errors.New("order must have at least one item")
           ErrDuplicateOrder   = errors.New("duplicate order") // not returned to client
       )

       type Service struct {
           db      *sqlc.Queries
           pool    *pgxpool.Pool
           invSvc  *inventory.Service
       }

       func NewService(pool *pgxpool.Pool, invSvc *inventory.Service) *Service {
           return &Service{
               db:     sqlc.New(pool),
               pool:   pool,
               invSvc: invSvc,
           }
       }

       // OrderItemInput for creating an order line
       type OrderItemInput struct {
           VariantID     string
           Quantity      int32
           PriceSnapshot int64 // in satang
       }

       // CreateOrderInput for creating an order
       type CreateOrderInput struct {
           ID     string // client-generated UUID
           UserID string
           Items  []OrderItemInput
       }

       // OrderResult returned after creation
       type OrderResult struct {
           ID        string
           UserID    string
           Status    string
           Total     int64
           CreatedAt time.Time
       }

       // CreateOrder creates a single order with inventory deduction.
       // Idempotent: if order ID already exists, returns the existing order without error.
       func (s *Service) CreateOrder(ctx context.Context, input CreateOrderInput) (OrderResult, error) {
           // Validate
           if input.ID == "" {
               return OrderResult{}, fmt.Errorf("%w: order id is required", ErrInvalidOrder)
           }
           if input.UserID == "" {
               return OrderResult{}, fmt.Errorf("%w: user id is required", ErrInvalidOrder)
           }
           if len(input.Items) == 0 {
               return OrderResult{}, ErrEmptyOrder
           }

           var orderUUID pgtype.UUID
           if err := orderUUID.Scan(input.ID); err != nil {
               return OrderResult{}, fmt.Errorf("invalid order id: %w", err)
           }

           var userUUID pgtype.UUID
           if err := userUUID.Scan(input.UserID); err != nil {
               return OrderResult{}, fmt.Errorf("invalid user id: %w", err)
           }

           // Check if order already exists (idempotency)
           exists, err := s.db.CheckOrderExists(ctx, orderUUID)
           if err != nil {
               return OrderResult{}, fmt.Errorf("checking order existence: %w", err)
           }
           if exists {
               // Return existing order
               existing, err := s.db.GetOrder(ctx, orderUUID)
               if err != nil {
                   return OrderResult{}, fmt.Errorf("getting existing order: %w", err)
               }
               return OrderResult{
                   ID:        existing.ID.String(),
                   UserID:    existing.UserID.String(),
                   Status:    existing.Status,
                   Total:     existing.Total,
                   CreatedAt: existing.CreatedAt.Time,
               }, nil
           }

           // Calculate total and validate items
           var total int64
           for _, item := range input.Items {
               if item.Quantity <= 0 {
                   return OrderResult{}, fmt.Errorf("%w: quantity must be positive", ErrInvalidOrder)
               }
               if item.PriceSnapshot < 0 {
                   return OrderResult{}, fmt.Errorf("%w: price cannot be negative", ErrInvalidOrder)
               }
               total += int64(item.Quantity) * item.PriceSnapshot
           }

           // Use a transaction to ensure atomicity: order + items + inventory deduction
           tx, err := s.pool.Begin(ctx)
           if err != nil {
               return OrderResult{}, fmt.Errorf("beginning transaction: %w", err)
           }
           defer tx.Rollback(ctx)

           txQueries := sqlc.New(tx)

           // Insert order
           err = txQueries.CreateOrder(ctx, sqlc.CreateOrderParams{
               ID:     orderUUID,
               UserID: userUUID,
               Status: "completed",
               Total:  total,
           })
           if err != nil {
               return OrderResult{}, fmt.Errorf("creating order: %w", err)
           }

           // Insert order items and deduct inventory
           for _, item := range input.Items {
               var variantUUID pgtype.UUID
               if err := variantUUID.Scan(item.VariantID); err != nil {
                   return OrderResult{}, fmt.Errorf("invalid variant id: %w", err)
               }

               subtotal := int64(item.Quantity) * item.PriceSnapshot

               err = txQueries.CreateOrderItem(ctx, sqlc.CreateOrderItemParams{
                   OrderID:       orderUUID,
                   VariantID:     variantUUID,
                   Quantity:      item.Quantity,
                   PriceSnapshot: item.PriceSnapshot,
                   Subtotal:      subtotal,
               })
               if err != nil {
                   return OrderResult{}, fmt.Errorf("creating order item: %w", err)
               }

               // Deduct stock via inventory service (delta operation)
               _, err = s.invSvc.DeductStock(ctx, item.VariantID, int64(item.Quantity), input.ID)
               if err != nil {
                   return OrderResult{}, fmt.Errorf("deducting stock for variant %s: %w", item.VariantID, err)
               }
           }

           if err := tx.Commit(ctx); err != nil {
               return OrderResult{}, fmt.Errorf("committing transaction: %w", err)
           }

           return OrderResult{
               ID:        input.ID,
               UserID:    input.UserID,
               Status:    "completed",
               Total:     total,
               CreatedAt: time.Now(),
           }, nil
       }

       // SyncOrdersResult reports sync outcome
       type SyncOrdersResult struct {
           Processed int
           Failed    int
           Errors    []SyncError
       }

       type SyncError struct {
           OrderID string
           Error   string
       }

       // SyncOrders processes a batch of offline orders with individual error handling.
       // Each order is processed independently — failure of one does not roll back others.
       func (s *Service) SyncOrders(ctx context.Context, userID string, orders []CreateOrderInput) (SyncOrdersResult, error) {
           result := SyncOrdersResult{}

           for _, input := range input {
               // Override userID from authenticated context (do not trust client-provided userID for sync)
               input.UserID = userID

               _, err := s.CreateOrder(ctx, input)
               if err != nil {
                   result.Failed++
                   result.Errors = append(result.Errors, SyncError{
                       OrderID: input.ID,
                       Error:   err.Error(),
                   })
                   // Continue processing other orders
                   continue
               }
               result.Processed++
           }

           return result, nil
       }

       // GetOrder retrieves an order with its items
       func (s *Service) GetOrder(ctx context.Context, orderID string) (OrderResult, []sqlc.GetOrderItemsRow, error) {
           var orderUUID pgtype.UUID
           if err := orderUUID.Scan(orderID); err != nil {
               return OrderResult{}, nil, fmt.Errorf("invalid order id: %w", err)
           }

           order, err := s.db.GetOrder(ctx, orderUUID)
           if err != nil {
               return OrderResult{}, nil, ErrOrderNotFound
           }

           items, err := s.db.GetOrderItems(ctx, orderUUID)
           if err != nil {
               return OrderResult{}, nil, fmt.Errorf("getting order items: %w", err)
           }

           return OrderResult{
               ID:        order.ID.String(),
               UserID:    order.UserID.String(),
               Status:    order.Status,
               Total:     order.Total,
               CreatedAt: order.CreatedAt.Time,
           }, items, nil
       }
       ```

       package sales

       import (
           "encoding/json"
           "net/http"

           "github.com/go-chi/chi/v5"
           "github.com/zestzero/openpos/internal/middleware"
       )

       type Handler struct {
           service *Service
       }

       func NewHandler(service *Service) *Handler {
           return &Handler{service: service}
       }

       func (h *Handler) Routes() http.Handler {
           r := chi.NewRouter()

           r.Post("/orders", h.CreateOrder)
           r.Post("/orders/sync", h.SyncOrders)
           r.Get("/orders/{id}", h.GetOrder)

           return r
       }

       type createOrderRequest struct {
           ID    string            `json:"id"`
           Items []orderItemRequest `json:"items"`
       }

       type orderItemRequest struct {
           VariantID     string `json:"variant_id"`
           Quantity      int32  `json:"quantity"`
           PriceSnapshot int64  `json:"price_snapshot"`
       }

       type successResponse struct {
           Data interface{} `json:"data"`
       }

       type errorResponse struct {
           Error string `json:"error"`
       }

       // POST /api/orders
       func (h *Handler) CreateOrder(w http.ResponseWriter, r *http.Request) {
           var req createOrderRequest
           if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
               http.Error(w, "invalid request body", http.StatusBadRequest)
               return
           }

           if req.ID == "" {
               http.Error(w, "order id is required", http.StatusBadRequest)
               return
           }
           if len(req.Items) == 0 {
               http.Error(w, "order must have at least one item", http.StatusBadRequest)
               return
           }

           userID := middleware.GetUserID(r.Context())
           if userID == "" {
               http.Error(w, "unauthorized", http.StatusUnauthorized)
               return
           }

           items := make([]OrderItemInput, len(req.Items))
           for i, item := range req.Items {
               items[i] = OrderItemInput{
                   VariantID:     item.VariantID,
                   Quantity:      item.Quantity,
                   PriceSnapshot: item.PriceSnapshot,
               }
           }

           order, err := h.service.CreateOrder(r.Context(), CreateOrderInput{
               ID:     req.ID,
               UserID: userID,
               Items:  items,
           })
           if err != nil {
               switch {
               case errors.Is(err, ErrEmptyOrder), errors.Is(err, ErrInvalidOrder):
                   http.Error(w, err.Error(), http.StatusBadRequest)
               default:
                   http.Error(w, err.Error(), http.StatusInternalServerError)
               }
               return
           }

           w.Header().Set("Content-Type", "application/json")
           w.WriteHeader(http.StatusCreated)
           json.NewEncoder(w).Encode(successResponse{Data: order})
       }

       // POST /api/orders/sync
       func (h *Handler) SyncOrders(w http.ResponseWriter, r *http.Request) {
           var req struct {
               Orders []createOrderRequest `json:"orders"`
           }
           if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
               http.Error(w, "invalid request body", http.StatusBadRequest)
               return
           }

           userID := middleware.GetUserID(r.Context())
           if userID == "" {
               http.Error(w, "unauthorized", http.StatusUnauthorized)
               return
           }

           orders := make([]CreateOrderInput, len(req.Orders))
           for i, o := range req.Orders {
               items := make([]OrderItemInput, len(o.Items))
               for j, item := range o.Items {
                   items[j] = OrderItemInput{
                       VariantID:     item.VariantID,
                       Quantity:      item.Quantity,
                       PriceSnapshot: item.PriceSnapshot,
                   }
               }
               orders[i] = CreateOrderInput{
                   ID:     o.ID,
                   UserID: userID,
                   Items:  items,
               }
           }

           result, err := h.service.SyncOrders(r.Context(), userID, orders)
           if err != nil {
               http.Error(w, err.Error(), http.StatusInternalServerError)
               return
           }

           w.Header().Set("Content-Type", "application/json")
           json.NewEncoder(w).Encode(successResponse{Data: result})
       }

       // GET /api/orders/{id}
       func (h *Handler) GetOrder(w http.ResponseWriter, r *http.Request) {
           id := chi.URLParam(r, "id")
           if id == "" {
               http.Error(w, "order id is required", http.StatusBadRequest)
               return
           }

           order, items, err := h.service.GetOrder(r.Context(), id)
           if err != nil {
               if errors.Is(err, ErrOrderNotFound) {
                   http.Error(w, "order not found", http.StatusNotFound)
                   return
               }
               http.Error(w, err.Error(), http.StatusInternalServerError)
               return
           }

           type orderItemResponse struct {
               VariantID     string `json:"variant_id"`
               VariantName   string `json:"variant_name"`
               Quantity      int32  `json:"quantity"`
               PriceSnapshot int64  `json:"price_snapshot"`
               Subtotal      int64  `json:"subtotal"`
           }

           itemResponses := make([]orderItemResponse, len(items))
           for i, item := range items {
               itemResponses[i] = orderItemResponse{
                   VariantID:     item.VariantID.String(),
                   VariantName:   item.VariantName,
                   Quantity:      item.Quantity,
                   PriceSnapshot: item.PriceSnapshot,
                   Subtotal:      item.Subtotal,
               }
           }

           w.Header().Set("Content-Type", "application/json")
           json.NewEncoder(w).Encode(successResponse{Data: struct {
               Order OrderResult         `json:"order"`
               Items []orderItemResponse `json:"items"`
           }{
               Order: order,
               Items: itemResponses,
           }})
       }
       ```
